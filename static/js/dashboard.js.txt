function metersToLatLng(center, eastMeters, northMeters) {
    const lat = center[0];
    const lng = center[1];

    const newLat = lat + (northMeters / 111320);
    const newLng = lng + (eastMeters / (111320 * Math.cos(lat * Math.PI / 180)));

    return [newLat, newLng];
}

function addHatchedZone(center, radiusMeters, color, fillColor, label, labelClass) {
    const layerGroup = L.layerGroup().addTo(map);

    const circle = L.circle(center, {
        radius: radiusMeters,
        color: color,
        fillColor: fillColor,
        fillOpacity: 0.20,
        weight: 2
    }).addTo(layerGroup);

    circle.bindTooltip(label, {
        permanent: true,
        direction: "center",
        className: `map-label ${labelClass}`
    });

    const slope = 0.55;
    const step = 35;

    for (let b = -radiusMeters; b <= radiusMeters; b += step) {
        const A = 1 + slope * slope;
        const B = 2 * slope * b;
        const C = b * b - radiusMeters * radiusMeters;
        const disc = B * B - 4 * A * C;

        if (disc < 0) continue;

        const x1 = (-B - Math.sqrt(disc)) / (2 * A);
        const x2 = (-B + Math.sqrt(disc)) / (2 * A);

        const y1 = slope * x1 + b;
        const y2 = slope * x2 + b;

        const p1 = metersToLatLng(center, x1, y1);
        const p2 = metersToLatLng(center, x2, y2);

        L.polyline([p1, p2], {
            color: color,
            weight: 1.4,
            opacity: 0.75
        }).addTo(layerGroup);
    }

    return circle;
}

function startDroneAnimation(routeCoords) {
    if (droneAnimationInterval) {
        clearInterval(droneAnimationInterval);
        droneAnimationInterval = null;
    }

    const droneIcon = L.divIcon({
        className: "",
        html: `<div class="drone-marker">âœˆ</div>`,
        iconSize: [36, 36],
        iconAnchor: [18, 18],
        popupAnchor: [0, -18]
    });

    droneMarker = L.marker(routeCoords[0], {
        icon: droneIcon
    })
    .addTo(map)
    .bindPopup("Ä°HA Konumu");

    let segment = 0;
    let t = 0;

    droneAnimationInterval = setInterval(() => {
        if (!droneMarker || !map) return;

        const start = routeCoords[segment];
        const end = routeCoords[(segment + 1) % routeCoords.length];

        const lat = start[0] + (end[0] - start[0]) * t;
        const lng = start[1] + (end[1] - start[1]) * t;

        droneMarker.setLatLng([lat, lng]);

        t += 0.035;

        if (t >= 1) {
            t = 0;
            segment = (segment + 1) % routeCoords.length;
        }
    }, 120);
}

function updateChart(d) {
    const movement = Number(d.movement || 0);
    const threshold = Number(d.threshold || 0);

    chartLabels.push(new Date().toLocaleTimeString("tr-TR"));
    movementData.push(movement);
    thresholdData.push(threshold);

    if (chartLabels.length > 60) {
        chartLabels.shift();
        movementData.shift();
        thresholdData.shift();
    }

    drawCanvasChart();
}

function drawCanvasChart() {
    const canvas = document.getElementById("movementChart");
    if (!canvas) return;

    const parent = canvas.parentElement;
    const width = parent ? parent.clientWidth : 500;
    const height = parent ? parent.clientHeight : 300;

    canvas.width = width;
    canvas.height = height;

    const ctx = canvas.getContext("2d");

    ctx.fillStyle = "#ffffff";
    ctx.fillRect(0, 0, width, height);

    const left = 45;
    const right = 15;
    const top = 20;
    const bottom = 35;

    const graphW = width - left - right;
    const graphH = height - top - bottom;

    const allValues = movementData.concat(thresholdData);
    let maxVal = Math.max(...allValues, 1.5);
    maxVal = Math.ceil(maxVal * 1.3 * 10) / 10;

    ctx.strokeStyle = "#d8dee9";
    ctx.lineWidth = 1;
    ctx.font = "12px Arial";
    ctx.fillStyle = "#64748b";

    for (let i = 0; i <= 5; i++) {
        const y = top + (graphH / 5) * i;
        const value = maxVal - (maxVal / 5) * i;

        ctx.beginPath();
        ctx.moveTo(left, y);
        ctx.lineTo(width - right, y);
        ctx.stroke();

        ctx.fillText(value.toFixed(1), 8, y + 4);
    }

    ctx.strokeStyle = "#94a3b8";
    ctx.beginPath();
    ctx.moveTo(left, top);
    ctx.lineTo(left, height - bottom);
    ctx.lineTo(width - right, height - bottom);
    ctx.stroke();

    function getX(index) {
        if (movementData.length <= 1) return left;
        return left + (index / (movementData.length - 1)) * graphW;
    }

    function getY(value) {
        return top + graphH - (value / maxVal) * graphH;
    }

    function drawLine(data, color, dashed) {
        if (data.length < 2) return;

        ctx.beginPath();
        ctx.strokeStyle = color;
        ctx.lineWidth = 3;
        ctx.setLineDash(dashed ? [8, 6] : []);

        data.forEach((value, index) => {
            const x = getX(index);
            const y = getY(value);

            if (index === 0) ctx.moveTo(x, y);
            else ctx.lineTo(x, y);
        });

        ctx.stroke();
        ctx.setLineDash([]);
    }

    drawLine(thresholdData, "#2563eb", true);
    drawLine(movementData, "#ef4444", false);

    if (movementData.length > 0) {
        const lastIndex = movementData.length - 1;
        const x = getX(lastIndex);
        const y = getY(movementData[lastIndex]);

        ctx.fillStyle = "#ef4444";
        ctx.beginPath();
        ctx.arc(x, y, 5, 0, Math.PI * 2);
        ctx.fill();

        ctx.fillStyle = "#111827";
        ctx.font = "bold 13px Arial";
        ctx.fillText("M: " + movementData[lastIndex].toFixed(2), Math.max(50, x - 70), y - 12);
    }

    ctx.font = "13px Arial";

    ctx.fillStyle = "#ef4444";
    ctx.fillRect(left, height - 18, 16, 4);
    ctx.fillStyle = "#111827";
    ctx.fillText("Hareket", left + 22, height - 13);

    ctx.fillStyle = "#2563eb";
    ctx.fillRect(left + 105, height - 18, 16, 4);
    ctx.fillStyle = "#111827";
    ctx.fillText("EÅŸik", left + 127, height - 13);

    if (movementData.length === 0) {
        ctx.fillStyle = "#64748b";
        ctx.font = "bold 16px Arial";
        ctx.fillText("Grafik veri bekliyor...", left + 20, top + 40);
    }
}

function updateThermal(d) {
    const grid = document.getElementById("thermalGrid");
    if (!grid) return;

    grid.innerHTML = "";

    const thermal = d.thermal || [];
    const minT = Number(d.thermal_min || 0);
    const maxT = Number(d.thermal_max || 0);

    if (!thermal.length) {
        grid.innerHTML = "<div>Termal veri yok</div>";
        return;
    }

    thermal.forEach(row => {
        row.forEach(temp => {
            const cell = document.createElement("div");
            cell.className = "thermal-cell";

            let ratio = 0;

            if (maxT > minT) {
                ratio = (temp - minT) / (maxT - minT);
            }

            const r = Math.floor(255 * ratio);
            const b = Math.floor(255 * (1 - ratio));
            const g = Math.floor(110 * (1 - Math.abs(ratio - 0.5)));

            cell.style.backgroundColor = `rgb(${r}, ${g}, ${b})`;
            cell.innerText = Number(temp).toFixed(1);

            grid.appendChild(cell);
        });
    });
}

function sendServoValue(type) {
    let slider = null;
    let endpoint = "";

    if (type === "pan") {
        slider = document.getElementById("panSlider");
        if (!slider) return;
        endpoint = `/servo/pan/${slider.value}`;
    }

    if (type === "tilt") {
        slider = document.getElementById("tiltSlider");
        if (!slider) return;
        endpoint = `/servo/tilt/${slider.value}`;
    }

    if (endpoint === "") {
        return;
    }

    fetch(endpoint, {
        method: "POST"
    })
    .then(response => response.json())
    .then(data => {
        if (!data.ok) {
            console.error("Servo komut hatasÄ±:", data.message);
        }
    })
    .catch(error => {
        console.error("Servo gÃ¶nderme hatasÄ±:", error);
    });
}

let cameraZoomIndex = 0;
const cameraZoomLevels = [1, 1.25, 1.5, 2, 2.5, 3, 4, 5, 6, 7, 8, 9, 10];

    const cameraFeed = document.getElementById("cameraFeed");

    if (resetBtn) {

    applyCameraZoom();

/* =========================================================
   DETAS CAMERA FIXED PANEL + SEPARATE SCAN CARD + LIMIT LOGS
   ========================================================= */

(function () {
    const DETAS_MAX_DETECTIONS = 5;

    function detasFindDetectionList() {
        return document.getElementById("detectionsList")
            || document.querySelector(".detections-list")
            || document.querySelector(".detection-list");
    }

    function detasFindCameraPanel() {
        return document.querySelector(".camera-panel");
    }

    function detasFindScanPanel() {
        return document.getElementById("cameraScanPanel")
            || document.querySelector(".camera-scan-panel");
    }

    function detasEnsureCameraViewport() {
        const img = document.getElementById("cameraFeed");
        if (!img) return;

        if (!img.parentElement.classList.contains("camera-viewport")) {
            const wrap = document.createElement("div");
            wrap.className = "camera-viewport";
            img.parentElement.insertBefore(wrap, img);
            wrap.appendChild(img);
        }
    }

    function detasEnsureDetectionWrap() {
        const cameraPanel = detasFindCameraPanel();
        const detectionList = detasFindDetectionList();
        if (!cameraPanel || !detectionList) return;

        detectionList.classList.add("camera-detections-list");

        if (!detectionList.parentElement.classList.contains("camera-detections-wrap")) {
            const wrap = document.createElement("div");
            wrap.className = "camera-detections-wrap";

            const title = document.createElement("div");
            title.className = "camera-detections-title";
            title.innerText = "AlgÄ±lama KayÄ±tlarÄ±";

            detectionList.parentElement.insertBefore(wrap, detectionList);
            wrap.appendChild(title);
            wrap.appendChild(detectionList);
        }
    }

    function detasLimitDetections() {
        const detectionList = detasFindDetectionList();
        if (!detectionList) return;

        const items = Array.from(detectionList.children).filter(el => el.nodeType === 1);

        items.forEach((item, index) => {
            item.style.display = index < DETAS_MAX_DETECTIONS ? "" : "none";
        });
    }

    function detasMoveScanPanelOutsideCamera() {
        const cameraPanel = detasFindCameraPanel();
        const scanPanel = detasFindScanPanel();
        if (!cameraPanel || !scanPanel) return;

        const parent = cameraPanel.parentElement;
        if (!parent) return;

        scanPanel.classList.add("detached-scan-card");

        if (cameraPanel.nextElementSibling !== scanPanel || scanPanel.parentElement !== parent) {
            parent.insertBefore(scanPanel, cameraPanel.nextSibling);
        }
    }

    function detasInitCameraStableLayout() {
        detasEnsureCameraViewport();
        detasEnsureDetectionWrap();
        detasLimitDetections();
        detasMoveScanPanelOutsideCamera();
    }

    function detasObserveDetectionChanges() {
        const detectionList = detasFindDetectionList();
        if (!detectionList || detectionList.dataset.observerAttached === "1") return;

        const observer = new MutationObserver(() => {
            detasLimitDetections();
        });

        observer.observe(detectionList, {
            childList: true,
            subtree: false
        });

        detectionList.dataset.observerAttached = "1";
    }

    document.addEventListener("DOMContentLoaded", function () {
        detasInitCameraStableLayout();
        detasObserveDetectionChanges();

        setTimeout(() => {
            detasInitCameraStableLayout();
            detasObserveDetectionChanges();
        }, 400);

        setTimeout(() => {
            detasInitCameraStableLayout();
            detasObserveDetectionChanges();
        }, 1200);
    });

    window.detasRefreshCameraLayout = function () {
        detasInitCameraStableLayout();
        detasObserveDetectionChanges();
    };
})();




/* =========================================================
   DETAS FINAL CONTENT BOX LAYOUT JS
   Mevcut DOM'u sidebar + content-box yapÄ±sÄ±na alÄ±r
   ========================================================= */
(function () {
    function txt(el) {
        return (el.innerText || el.textContent || "").trim();
    }

    function findSidebar() {
        const candidates = Array.from(document.querySelectorAll(
            "aside, .sidebar, .side-panel, .left-sidebar, .detas-sidebar-fixed, nav"
        ));

        let found = candidates.find(el => {
            const t = txt(el);
            return t.includes("DETAS") && t.includes("Genel Durum");
        });

        if (!found) {
            found = candidates.find(el => {
                const t = txt(el);
                return t.includes("Genel Durum") && t.includes("CanlÄ± Kamera");
            });
        }

        if (found && found.tagName && found.tagName.toLowerCase() === "nav") {
            let p = found.parentElement;
            while (p && p !== document.body) {
                const t = txt(p);
                if (t.includes("DETAS") || t.includes("Sistem Durumu")) {
                    return p;
                }
                p = p.parentElement;
            }
        }

        return found;
    }

    function isSkippable(el) {
        if (!el || !el.tagName) return true;

        const tag = el.tagName.toLowerCase();

        if (["script", "style", "link", "meta", "title"].includes(tag)) return true;
        if (el.id === "detasSidebarFinal") return true;
        if (el.id === "detasMainAreaFinal") return true;

        return false;
    }

    function buildFinalLayout() {
        const sidebar = findSidebar();
        if (!sidebar) return;

        sidebar.id = "detasSidebarFinal";
        sidebar.classList.add("detas-sidebar-final");

        let mainArea = document.getElementById("detasMainAreaFinal");
        let contentBox = document.getElementById("detasContentBoxFinal");

        if (!mainArea) {
            mainArea = document.createElement("main");
            mainArea.id = "detasMainAreaFinal";
        }

        if (!contentBox) {
            contentBox = document.createElement("div");
            contentBox.id = "detasContentBoxFinal";
            mainArea.appendChild(contentBox);
        }

        if (sidebar.parentElement !== document.body) {
            document.body.insertBefore(sidebar, document.body.firstChild);
        }

        if (mainArea.parentElement !== document.body) {
            document.body.appendChild(mainArea);
        }

        const bodyChildren = Array.from(document.body.children);

        bodyChildren.forEach(child => {
            if (isSkippable(child)) return;
            if (child === contentBox) return;
            if (child.contains && child.contains(sidebar)) return;

            contentBox.appendChild(child);
        });

        // Eski shell iÃ§erde kaldÄ±ysa geniÅŸlik bozmasÄ±n
        const oldShell = contentBox.querySelector("#detasAppShell");
        if (oldShell) {
            oldShell.style.display = "block";
            oldShell.style.width = "100%";
            oldShell.style.minHeight = "auto";
        }
    }

    function fixCameraButtons() {
        const content = document.getElementById("detasContentBoxFinal") || document.body;

        const buttons = Array.from(content.querySelectorAll("button")).filter(btn => {
            const t = txt(btn);
            return (
                t === "CAM0 V3" ||
                t === "CAM1 Gece" ||
                t.startsWith("Zoom -") ||
                t.startsWith("Zoom +") ||
                /^Zoom\s+\d+(\.\d+)?x$/i.test(t)
            );
        });

        if (!buttons.length) return;

        const cameraPanel =
            content.querySelector(".camera-panel") ||
            Array.from(content.querySelectorAll("div, section, article")).find(el => txt(el).includes("CanlÄ± Kamera + YOLO"));

        if (!cameraPanel) return;

        let toolbar = cameraPanel.querySelector(".detas-camera-toolbar");
        if (!toolbar) {
            toolbar = document.createElement("div");
            toolbar.className = "detas-camera-toolbar";

            const viewport =
                cameraPanel.querySelector(".camera-viewport") ||
                cameraPanel.querySelector("#cameraFeed");

            if (viewport) {
                viewport.parentElement.insertBefore(toolbar, viewport);
            } else {
                cameraPanel.insertBefore(toolbar, cameraPanel.children[2] || cameraPanel.firstChild);
            }
        }

        const cam0 = buttons.find(b => txt(b) === "CAM0 V3");
        const cam1 = buttons.find(b => txt(b) === "CAM1 Gece");
        const zmn  = buttons.find(b => txt(b).startsWith("Zoom -"));
        const zmp  = buttons.find(b => txt(b).startsWith("Zoom +"));
        const zmi  = buttons.find(b => /^Zoom\s+\d+(\.\d+)?x$/i.test(txt(b)));

        [cam0, cam1, zmn, zmi, zmp].filter(Boolean).forEach(btn => {
            toolbar.appendChild(btn);
        });
    }

    function runFinalLayout() {
        buildFinalLayout();
        fixCameraButtons();
    }

    document.addEventListener("DOMContentLoaded", function () {
        runFinalLayout();
        setTimeout(runFinalLayout, 300);
        setTimeout(runFinalLayout, 1000);
        setTimeout(runFinalLayout, 2000);
    });

    window.detasFinalLayoutFix = runFinalLayout;
})();
